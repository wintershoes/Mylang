#词法规则冒号前写词法的类型，冒号后写匹配规则，匹配规则用单引号括起来

# Conditional Key Words
IF          :'if';
ELSE        :'else';

# Move Interfaces
FORWARD     :'forward';
BACKWARD    :'backward';

# Rotate Interfaces
TURNRIGHT   :'turnright';
TURNLEFT    :'turnleft';

# Naviagaton Interface
GOTO        :'goto';
BACK_TO_USER:'back_to_user';

# Approach 'object' Interface
APPROACH    :'approach';
# Grasp 'object' Interface
GRASP       :'grasp';

# Conditional expression function
EXIST       :'exist';

# 符号也按如下规则识别，符号类必须在前面加个SYMBOL
# 特殊符号允许和其他字符可以不用空格隔开(会预处理用空格隔开特殊符号，之后再继续分词)，但是其他单词之间必须空格隔开
# 特殊符号按照最长优先匹配规则，也就是!=会优先!匹配，==会优先=被匹配
SYMBOL LBRACE      : '{';
SYMBOL RBRACE      : '}';
SYMBOL COMMA       : ',';
SYMBOL SEMI        : ';';

# Boolean Symbols
SYMBOL NOT         :'!';
SYMBOL AND         :'&';
SYMBOL OR          :'|';

#  标识符和数字按如下规则识别，不用自定义
#ID          : [a-zA-Z_][a-zA-Z_0-9]*;
#NUMBER      : '-'?[0-9]+ ('.' [0-9]+)?;
# 空格、换行符和注释默认省略
# 词法分析器按空格来分词
