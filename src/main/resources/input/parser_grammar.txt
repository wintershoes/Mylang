# 不要写含有左递归、间接左公因子的语法。
# 最好通过改写文法的方法，保证是LL1语法。
program         : statement+;
statement       : command | condstatement;

condstatement   : ifStatement;
ifStatement     : IF boolExp LBRACE Case RBRACE elseStatement;
elseStatement   : ELSE ifStatement | ELSE LBRACE Case RBRACE;
Case            : statement+;
boolExp         : EXIST ID;

# list all command types.
command         : (forwardCommand | backwardCommand | turnrightCommand | turnleftCommand|
gotoCommand | approachCommand | graspCommand | backCommand ) SEMI;
forwardCommand  : FORWARD NUMBER;
backwardCommand : BACKWARD NUMBER;
turnrightCommand: TURNRIGHT NUMBER;
turnleftCommand : TURNLEFT NUMBER;
gotoCommand     : GOTO NUMBER COMMA NUMBER;
approachCommand : APPROACH ID;
graspCommand    : GRASP ID;
backCommand     : BACK_TO_USER;
